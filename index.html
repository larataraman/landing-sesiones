function onFormSubmit(e) { const response = e.namedValues; const nombre = response["Nombre"][0]; // Campo en tu formulario const email = response["Email"][0]; const fecha = response["Fecha"][0]; // Formato YYYY-MM-DD const hora = response["Horario"][0]; // Formato HH:mm // Crear fecha y hora de inicio y fin const startTime = new Date(`${fecha}T${hora}:00`); const endTime = new Date(startTime.getTime() + 60*60*1000); // 1 hora para individual if (hora.includes("grupal")) { endTime.setMinutes(endTime.getMinutes() + 30); // 1:30 para grupal } // Crear evento en tu calendario principal const calendar = CalendarApp.getCalendarById("larataraman@gmail.com"); const event = calendar.createEvent( `Sesión con ${nombre}`, startTime, endTime, {guests: email, sendInvites: true} ); // Generar Google Meet event.setGuestsCanModify(false); event.addConference("hangoutsMeet"); // Enviar mail de confirmación const meetLink = event.getHangoutLink(); MailApp.sendEmail({ to: email, subject: "Confirmación de tu sesión", htmlBody: ` Hola ${nombre},

Tu sesión quedó reservada para ${fecha} a las ${hora}.
Aquí tu enlace de Google Meet: ${meetLink}

Gracias. ` }); }
